// PASTIKAN LINK URL BERIKUT ADALAH HASIL DEPLOY VERSI BARU SPREADSHEET ANDA const SHEET_WEBAPP_URL = "https://diansulistyo67manutd-design.github.io/kas-cluster-dago/"; const loading = document.getElementById('loading'); const loginPage = document.getElementById('login-page'); const wargaPage = document.getElementById('warga-page'); const adminPage = document.getElementById('admin-page'); function toggleLoading(show) { if(show) loading.classList.remove('hidden'); else loading.classList.add('hidden'); } // Mesin utama pengirim data anti Gagal / Anti-CORS browser function sendToGoogleSheet(params, callbackName) { const oldScript = document.getElementById('jsonp-script'); if (oldScript) oldScript.remove(); let url = SHEET_WEBAPP_URL + "?" + params + "&callback=" + callbackName + "&_ts=" + new Date().getTime(); const script = document.createElement('script'); script.id = 'jsonp-script'; script.src = url; document.body.appendChild(script); } // ========================================== // HANDLE FORM LOGIN // ========================================== document.getElementById('login-form').addEventListener('submit', function(e) { e.preventDefault(); toggleLoading(true); const user = encodeURIComponent(document.getElementById('username').value.trim()); const pass = encodeURIComponent(document.getElementById('password').value); sendToGoogleSheet(`action=login&username=${user}&password=${pass}`, 'handleLoginResponse'); }); function handleLoginResponse(response) { toggleLoading(false); if (response.status === "success") { loginPage.classList.add('hidden'); if (response.role === 'admin') { adminPage.classList.remove('hidden'); } else { wargaPage.classList.remove('hidden'); } alert("Berhasil masuk ke sistem!"); } else { alert("Gagal Login: " + response.message); } } // ========================================== // HANDLE FORM GANTI PASSWORD // ========================================== document.getElementById('change-password-form').addEventListener('submit', function(e) { e.preventDefault(); toggleLoading(true); const oldPass = encodeURIComponent(document.getElementById('old-pass').value); const newPass = encodeURIComponent(document.getElementById('new-pass').value); sendToGoogleSheet(`action=changePassword&oldPassword=${oldPass}&newPassword=${newPass}`, 'handleChangePasswordResponse'); }); function handleChangePasswordResponse(response) { toggleLoading(false); alert(response.message); // Menampilkan notifikasi langsung dari server Google Sheets Anda if (response.status === "success") { closeModal('password-modal'); document.getElementById('change-password-form').reset(); } } function logout() { adminPage.classList.add('hidden'); wargaPage.classList.add('hidden'); loginPage.classList.remove('hidden'); document.getElementById('login-form').reset(); } function openModal(id) { document.getElementById(id).classList.remove('hidden'); } function closeModal(id) { document.getElementById(id).classList.add('hidden'); }
